;---------- atoi ----------
atoi:
    xor rax,rax
    xor rcx,rcx
.next:
    movzx rdx,[rsi+rcx]
    cmp rdx,10
    je .done
    sub rdx,'0'
    imul rax,rax,10
    add rax,rdx
    inc rcx
    jmp .next
.done:
    ret

;---------- itoa ----------
itoa:
    mov rcx,10
    lea rbx,[rdi+19]
    mov byte[rbx],0
.convert:
    xor rdx,rdx
    div rcx
    add dl,'0'
    dec rbx
    mov [rbx],dl
    test rax,rax
    jnz .convert
    mov rsi,rbx
.copy:
    mov al,[rsi]
    mov [rdi],al
    inc rsi
    inc rdi
    test al,al
    jnz .copy
    ret

;---------- strlen ----------
strlen:
    xor rax,rax
.lp:
    cmp byte[rsi+rax],0
    je .done
    inc rax
    jmp .lp
.done:
    ret
