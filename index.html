// ------------------ PRINT NORMAL MATRIX ------------------
void printMatrix(int mat[][20], int r, int c)
{
    for (int i = 0; i < r; i++)
    {
        for (int j = 0; j < c; j++)
        {
            cout << mat[i][j] << " ";
        }
        cout << endl;
    }
}

// ------------------ CONVERT NORMAL TO SPARSE ------------------
void convertToSparse(int mat[][20], int r, int c, int sparse[][3])
{
    int k = 1;
    sparse[0][0] = r;
    sparse[0][1] = c;

    for (int i = 0; i < r; i++)
    {
        for (int j = 0; j < c; j++)
        {
            if (mat[i][j] != 0)
            {
                sparse[k][0] = i;
                sparse[k][1] = j;
                sparse[k][2] = mat[i][j];
                k++;
            }
        }
    }
    sparse[0][2] = k - 1;  // total non-zero elements
}

// ------------------ DISPLAY SPARSE ------------------
void displaySparse(int sparse[][3])
{
    int total = sparse[0][2];
    cout << "Row\tCol\tValue\n";
    for (int i = 0; i <= total; i++)
    {
        cout << sparse[i][0] << "\t" 
             << sparse[i][1] << "\t" 
             << sparse[i][2] << endl;
    }
}

// ------------------ SIMPLE TRANSPOSE ------------------
void simpleTranspose(int s[][3], int t[][3])
{
    int rows = s[0][0], cols = s[0][1], nonzero = s[0][2];
    t[0][0] = cols;
    t[0][1] = rows;
    t[0][2] = nonzero;

    int k = 1;
    for (int col = 0; col < cols; col++)
    {
        for (int i = 1; i <= nonzero; i++)
        {
            if (s[i][1] == col)
            {
                t[k][0] = s[i][1];
                t[k][1] = s[i][0];
                t[k][2] = s[i][2];
                k++;
            }
        }
    }
}

// ------------------ FAST TRANSPOSE ------------------
void fastTranspose(int s[][3], int t[][3])
{
    int rows = s[0][0], cols = s[0][1], nonzero = s[0][2];

    t[0][0] = cols;
    t[0][1] = rows;
    t[0][2] = nonzero;

    int rowterm[20] = {0};
    int startingPos[20];

    if (nonzero > 0)
    {
        // Count elements in each column
        for (int i = 1; i <= nonzero; i++)
            rowterm[s[i][1]]++;

        // Starting position for each column
        startingPos[0] = 1;
        for (int i = 1; i < cols; i++)
            startingPos[i] = startingPos[i - 1] + rowterm[i - 1];

        // Construct transpose
        for (int i = 1; i <= nonzero; i++)
        {
            int pos = startingPos[s[i][1]]++;
            t[pos][0] = s[i][1];
            t[pos][1] = s[i][0];
            t[pos][2] = s[i][2];
        }
    }
}

// ------------------ ADD TWO SPARSE MATRICES ------------------
void addSparse(int a[][3], int b[][3], int c[][3])
{
    if (a[0][0] != b[0][0] || a[0][1] != b[0][1])
    {
        cout << "Addition not possible.\n";
        return;
    }

    int i = 1, j = 1, k = 1;
    int totalA = a[0][2], totalB = b[0][2];

    c[0][0] = a[0][0];
    c[0][1] = a[0][1];

    while (i <= totalA && j <= totalB)
    {
        // Compare rows
        if (a[i][0] < b[j][0])
        {
            c[k][0] = a[i][0];
            c[k][1] = a[i][1];
            c[k][2] = a[i][2];
            i++; k++;
        }
        else if (a[i][0] > b[j][0])
        {
            c[k][0] = b[j][0];
            c[k][1] = b[j][1];
            c[k][2] = b[j][2];
            j++; k++;
        }
        else
        {
            // Same row → compare columns
            if (a[i][1] < b[j][1])
            {
                c[k][0] = a[i][0];
                c[k][1] = a[i][1];
                c[k][2] = a[i][2];
                i++; k++;
            }
            else if (a[i][1] > b[j][1])
            {
                c[k][0] = b[j][0];
                c[k][1] = b[j][1];
                c[k][2] = b[j][2];
                j++; k++;
            }
            else
            {
                // Same position
                int sum = a[i][2] + b[j][2];
                if (sum != 0)
                {
                    c[k][0] = a[i][0];
                    c[k][1] = a[i][1];
                    c[k][2] = sum;
                    k++;
                }
                i++; j++;
            }
        }
    }

    // Copy remaining elements
    while (i <= totalA)
    {
        c[k][0] = a[i][0];
        c[k][1] = a[i][1];
        c[k][2] = a[i][2];
        i++; k++;
    }

    while (j <= totalB)
    {
        c[k][0] = b[j][0];
        c[k][1] = b[j][1];
        c[k][2] = b[j][2];
        j++; k++;
    }

    c[0][2] = k - 1;
}

// ------------------ MULTIPLY SPARSE MATRICES ------------------
void multiplySparse(int a[][3], int b[][3], int c[][3])
{
    if (a[0][1] != b[0][0])
    {
        cout << "Multiplication not possible.\n";
        return;
    }

    int rowsA = a[0][0], colsB = b[0][1];
    int nonA = a[0][2], nonB = b[0][2];

    int res[20][20] = {0};

    // Multiply non-zero terms
    for (int i = 1; i <= nonA; i++)
    {
        for (int j = 1; j <= nonB; j++)
        {
            if (a[i][1] == b[j][0])
            {
                int r = a[i][0];
                int ccol = b[j][1];
                res[r][ccol] += a[i][2] * b[j][2];
            }
        }
    }

    // Convert result to sparse
    convertToSparse(res, rowsA, colsB, c);
}

// ---------- (a) Bubble Sort by Roll Number ----------
void sortByRollno(Student students[], int n) {
    for (int i = 0; i < n - 1; i++) {
        for (int j = 0; j < n - i - 1; j++) {
            if (students[j].rollNo > students[j + 1].rollNo) {
                swap(students[j], students[j + 1]);
            }
        }
    }
    cout << "\nStudents sorted by Roll Number (Ascending):";
    display(students, n);
}

// ---------- (b) Insertion Sort by Name (Alphabetically) ----------
void sortByName(Student students[], int n) {
    for (int i = 1; i < n; i++) {
        Student key = students[i];
        int j = i - 1;
        while (j >= 0 && students[j].name > key.name) {
            students[j + 1] = students[j];
            j--;
        }
        students[j + 1] = key;
    }
    cout << "\nStudents sorted Alphabetically by Name:";
    display(students, n);
}

// ---------- (c) Quick Sort by SGPA (Descending for Toppers) ----------
int partition(Student arr[], int low, int high) {
    float pivot = arr[high].sgpa;
    int i = low - 1;
    for (int j = low; j < high; j++) {
        if (arr[j].sgpa > pivot) { // descending
            i++;
            swap(arr[i], arr[j]);
        }
    }
    swap(arr[i + 1], arr[high]);
    return i + 1;
}

void quickSort(Student arr[], int low, int high) {
    if (low < high) {
        int pi = partition(arr, low, high);
        quickSort(arr, low, pi - 1);
        quickSort(arr, pi + 1, high);
    }
}

void displayToppers(Student students[], int n) {
    cout << "\nTop 10 Toppers (Sorted by SGPA):";
    quickSort(students, 0, n - 1);
    int limit = (n < 10) ? n : 10;
    display(students, limit);
}

// ---------- (d) Linear Search by SGPA ----------
void searchBySGPA(Student students[], int n, float keySGPA) {
    bool found = false;
    cout << "\nSearching for students with SGPA = " << keySGPA << "...\n";
    for (int i = 0; i < n; i++) {

            found = true;
        }
    }

}

// ---------- (e) Binary Search by Name (Iterative) ----------
int binarySearchByName(Student students[], int n, string keyName) {
    int low = 0, high = n - 1;
    while (low <= high) {
        int mid = (low + high) / 2;
        if (students[mid].name == keyName)
            return mid;
        else if (students[mid].name < keyName)
            low = mid + 1;
        else
            high = mid - 1;
    }
    return -1;
}

void searchByName(Student students[], int n, string keyName) {
    sortByName(students, n); // ensure list is sorted by name
    int index = binarySearchByName(students, n, keyName);
    if (index != -1) {
        cout << "\nStudent found:\n";
        cout << "Roll No: " << students[index].rollNo
             << "\nName: " << students[index].name
             << "\nSGPA: " << students[index].sgpa
             << "\nDepartment: " << students[index].department << endl;
    } else {
        cout << "\nStudent named '" << keyName << "' not found.\n";
    }
}

// ---------- (f) Fibonacci Search by Department ----------
int fibonacciSearchByDept(Student students[], int n, string keyDept) {
    int fib2 = 0;
    int fib1 = 1;
    int fib = fib2 + fib1;

    while (fib < n)
    {
        fib2 = fib1;
        fib1 = fib;
        fib = fib2 + fib1;
    }

    int offset = -1;

    while (fib > 1) {
        int i = min(offset + fib2, n - 1);

        if (students[i].department < keyDept) {
            fib = fib1;
            fib1 = fib2;
            fib2 = fib - fib1;
            offset = i;
        }
        else if (students[i].department > keyDept) {
            fib = fib2;
            fib1 = fib1 - fib2;
            fib2 = fib - fib1;
        }
        else
            return i;
    }

    if (fib1 && students[offset + 1].department == keyDept)
        return offset + 1;

    return -1;
}

void searchByDepartment(Student students[], int n, string dept) {
    sortByName(students, n); // sort so Fibonacci search works properly
    int index = fibonacciSearchByDept(students, n, dept);
    if (index != -1)

int precedence(char op){
    if(op == '^') return 3;
    if(op == '*' || op == '/') return 2;
    if(op == '+' || op == '-') return 1;
    return 0;
}

bool isOperator(char c){
    return (c=='+' || c=='-' || c=='*' || c=='/' || c=='^');
}

bool isRightAssociative(char c){
    return  c == '^';
}

string infixToPostfix(string infix){
    stack st;
    string result = "";

    for(char c: infix){
        if(isalnum(c)){
            result += c;
        }else if(c == '('){
            st.push(c);
        }else if(c == ')'){
            while( !st.empty() && st.top() != '('){
                result += st.top();
                st.pop();
            }
            st.pop();
        }else if(isOperator(c)){
            while(!st.empty() && isOperator(st.top())){

                bool cond1 =  precedence(st.top()) > precedence(c);
                bool cond2 = precedence(st.top()) == precedence(c) &&  !isRightAssociative(c);

                if(cond1 || cond2){
                        result += st.top();
                        st.pop();
                }else break;
            }
            st.push(c);
        }

    }
    while(!st.empty()){
        result += st.top();
        st.pop();
    }
    return result;
}
